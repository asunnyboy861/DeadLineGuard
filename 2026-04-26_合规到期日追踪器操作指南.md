```swift
import Foundation
import SwiftData

// MARK: - Compliance Item Model
@Model
final class ComplianceItem {
    @Attribute(.unique) var id: UUID
    var name: String
    var category: ComplianceCategory
    var dueDate: Date
    var recurrence: RecurrenceRule
    var responsibleParty: String
    var status: ItemStatus
    var notes: String
    var completedAt: Date?
    var createdAt: Date
    var updatedAt: Date
    
    // Relationship: one-to-many documents
    @Relationship(deleteRule: .cascade, inverse: \Document.item)
    var documents: [Document] = []
    
    // Relationship: one-to-many audit logs
    @Relationship(deleteRule: .cascade, inverse: \AuditLog.item)
    var auditLogs: [AuditLog] = []
    
    // Computed: is overdue
    var isOverdue: Bool {
        status != .completed && dueDate < Date()
    }
    
    // Computed: days until due
    var daysUntilDue: Int {
        Calendar.current.dateComponents([.day], from: Date(), to: dueDate).day ?? 0
    }
    
    init(name: String, category: ComplianceCategory, dueDate: Date, 
         recurrence: RecurrenceRule = .none, responsibleParty: String = "") {
        self.id = UUID()
        self.name = name
        self.category = category
        self.dueDate = dueDate
        self.recurrence = recurrence
        self.responsibleParty = responsibleParty
        self.status = .pending
        self.notes = ""
        self.completedAt = nil
        self.createdAt = Date()
        self.updatedAt = Date()
    }
}

// MARK: - Category Enum
enum ComplianceCategory: String, Codable, CaseIterable {
    case tax = "Tax"
    case license = "License"
    case report = "Report"
    case insurance = "Insurance"
    case employee = "Employee"
    case other = "Other"
    
    var icon: String {
        switch self {
        case .tax: return "dollarsign.circle.fill"
        case .license: return "licenseplate.fill"
        case .report: return "doc.text.fill"
        case .insurance: return "shield.fill"
        case .employee: return "person.2.fill"
        case .other: return "folder.fill"
        }
    }
    
    var color: String {
        switch self {
        case .tax: return "TaxBlue"
        case .license: return "LicensePurple"
        case .report: return "ReportTeal"
        case .insurance: return "InsuranceGreen"
        case .employee: return "EmployeeOrange"
        case .other: return "OtherGray"
        }
    }
}

// MARK: - Status Enum
enum ItemStatus: String, Codable {
    case pending = "Pending"
    case inProgress = "In Progress"
    case completed = "Completed"
    case overdue = "Overdue"
}

// MARK: - Recurrence Rule
enum RecurrenceRule: String, Codable, CaseIterable {
    case none = "One-time"
    case monthly = "Monthly"
    case quarterly = "Quarterly"
    case semiAnnual = "Semi-Annual"
    case annual = "Annual"
    case biennial = "Biennial"
    
    func nextDate(from date: Date) -> Date? {
        guard self != .none else { return nil }
        let calendar = Calendar.current
        switch self {
        case .monthly: return calendar.date(byAdding: .month, value: 1, to: date)
        case .quarterly: return calendar.date(byAdding: .month, value: 3, to: date)
        case .semiAnnual: return calendar.date(byAdding: .month, value: 6, to: date)
        case .annual: return calendar.date(byAdding: .year, value: 1, to: date)
        case .biennial: return calendar.date(byAdding: .year, value: 2, to: date)
        case .none: return nil
        }
    }
}
```

### 3.3 Notification Service

```swift
import UserNotifications

// MARK: - Notification Service
final class NotificationService {
    static let shared = NotificationService()
    
    private let center = UNUserNotificationCenter.current()
    
    // Reminder schedule configuration
    struct ReminderConfig {
        let daysBefore: [Int]  // e.g., [30, 14, 7, 1]
        
        static let `default` = ReminderConfig(daysBefore: [30, 14, 7, 1])
        static let free = ReminderConfig(daysBefore: [1])  // Free tier: only 1 day
    }
    
    // Request notification permission
    func requestAuthorization() async throws -> Bool {
        try await center.requestAuthorization(options: [.alert, .badge, .sound])
    }
    
    // Schedule multi-stage reminders for a compliance item
    func scheduleReminders(for item: ComplianceItem, config: ReminderConfig = .default) async {
        let content = UNMutableNotificationContent()
        content.title = "⏰ Deadline Approaching"
        content.body = "\(item.name) is due soon"
        content.sound = .default
        content.categoryIdentifier = "COMPLIANCE_REMINDER"
        content.userInfo = ["itemID": item.id.uuidString]
        
        for daysBefore in config.daysBefore {
            guard let triggerDate = Calendar.current.date(
                byAdding: .day, value: -daysBefore, to: item.dueDate
            ), triggerDate > Date() else { continue }
            
            let components = Calendar.current.dateComponents(
                [.year, .month, .day, .hour, .minute], from: triggerDate
            )
            let trigger = UNCalendarNotificationTrigger(
                dateMatching: components, repeats: false
            )
            
            // Unique identifier per reminder stage
            let identifier = "compliance_\(item.id.uuidString)_\(daysBefore)d"
            let request = UNNotificationRequest(
                identifier: identifier, content: content, trigger: trigger
            )
            
            try? await center.add(request)
        }
        
        // Schedule overdue escalation
        scheduleOverdueReminder(for: item)
    }
    
    // Overdue escalation notification
    private func scheduleOverdueReminder(for item: ComplianceItem) {
        let content = UNMutableNotificationContent()
        content.title = "🚨 OVERDUE"
        content.body = "\(item.name) is PAST DUE! Take action now."
        content.sound = .defaultCritical
        content.categoryIdentifier = "COMPLIANCE_OVERDUE"
        
        // Notify 1 hour after deadline
        guard let overdueDate = Calendar.current.date(
            byAdding: .hour, value: 1, to: item.dueDate
        ), overdueDate > Date() else { return }
        
        let components = Calendar.current.dateComponents(
            [.year, .month, .day, .hour, .minute], from: overdueDate
        )
        let trigger = UNCalendarNotificationTrigger(
            dateMatching: components, repeats: false
        )
        let request = UNNotificationRequest(
            identifier: "overdue_\(item.id.uuidString)", 
            content: content, trigger: trigger
        )
        try? center.add(request)
    }
    
    // Cancel all reminders for an item
    func cancelReminders(for item: ComplianceItem) {
        let identifiers = [30, 14, 7, 1].map { 
            "compliance_\(item.id.uuidString)_\($0)d" 
        } + ["overdue_\(item.id.uuidString)"]
        center.removePendingNotificationRequests(withIdentifiers: identifiers)
    }
    
    // Mark complete and auto-create next recurring item
    func markComplete(item: ComplianceItem, modelContext: ModelContext) {
        item.status = .completed
        item.completedAt = Date()
        
        // Add audit log
        let log = AuditLog(action: "Completed", details: "Marked as completed")
        log.item = item
        modelContext.insert(log)
        
        // Cancel reminders
        cancelReminders(for: item)
        
        // Auto-create next recurring item
        if let nextDate = item.recurrence.nextDate(from: item.dueDate) {
            let newItem = ComplianceItem(
                name: item.name,
                category: item.category,
                dueDate: nextDate,
                recurrence: item.recurrence,
                responsibleParty: item.responsibleParty
            )
            newItem.notes = item.notes
            modelContext.insert(newItem)
            
            // Schedule reminders for new item
            Task { await scheduleReminders(for: newItem) }
        }
        
        try? modelContext.save()
    }
}
```

### 3.4 Dashboard View (SwiftUI)

```swift
import SwiftUI
import SwiftData

struct DashboardView: View {
    @Query(sort: \ComplianceItem.dueDate) var items: [ComplianceItem]
    @Environment(\.modelContext) private var modelContext
    
    var overdueItems: [ComplianceItem] {
        items.filter { $0.isOverdue }
    }
    
    var upcomingItems: [ComplianceItem] {
        items.filter { !$0.isOverdue && $0.status != .completed }
            .sorted { $0.daysUntilDue < $1.daysUntilDue }
    }
    
    var body: some View {
        NavigationStack {
            ScrollView {
                VStack(spacing: 20) {
                    // Stats Cards
                    StatsRow(overdue: overdueItems.count, 
                             upcoming: upcomingItems.count,
                             total: items.count)
                    
                    // Overdue Section (Red Alert)
                    if !overdueItems.isEmpty {
                        SectionHeader(title: "🚨 Overdue", count: overdueItems.count, 
                                      color: .red)
                        ForEach(overdueItems) { item in
                            ComplianceItemCard(item: item, style: .overdue)
                        }
                    }
                    
                    // Upcoming Section
                    if !upcomingItems.isEmpty {
                        SectionHeader(title: "📅 Upcoming", count: upcomingItems.count,
                                      color: .blue)
                        ForEach(upcomingItems.prefix(5)) { item in
                            ComplianceItemCard(item: item, style: .upcoming)
                        }
                    }
                    
                    // Empty State
                    if items.isEmpty {
                        EmptyStateView()
                    }
                }
                .padding()
            }
            .navigationTitle("DeadLineGuard")
            .toolbar {
                ToolbarItem(placement: .topBarTrailing) {
                    Button(action: { showingAddSheet = true }) {
                        Image(systemName: "plus.circle.fill")
                    }
                }
            }
            .sheet(isPresented: $showingAddSheet) {
                AddComplianceItemView()
            }
        }
    }
    
    @State private var showingAddSheet = false
}

// MARK: - Compliance Item Card
struct ComplianceItemCard: View {
    let item: ComplianceItem
    let style: CardStyle
    
    enum CardStyle { case overdue, upcoming, completed }
    
    var body: some View {
        HStack(spacing: 12) {
            // Category Icon
            Image(systemName: item.category.icon)
                .font(.title2)
                .foregroundStyle(categoryColor)
                .frame(width: 44, height: 44)
                .background(categoryColor.opacity(0.15))
                .clipShape(RoundedRectangle(cornerRadius: 10))
            
            // Content
            VStack(alignment: .leading, spacing: 4) {
                Text(item.name)
                    .font(.headline)
                    .lineLimit(1)
                
                HStack {
                    Text(item.dueDate, style: .date)
                        .font(.subheadline)
                        .foregroundStyle(.secondary)
                    
                    if item.daysUntilDue <= 7 && item.daysUntilDue > 0 {
                        Text("\(item.daysUntilDue)d left")
                            .font(.caption.bold())
                            .foregroundStyle(.orange)
                    }
                }
                
                if !item.responsibleParty.isEmpty {
                    Label(item.responsibleParty, systemImage: "person.fill")
                        .font(.caption)
                        .foregroundStyle(.tertiary)
                }
            }
            
            Spacer()
            
            // Status Badge
            statusBadge
        }
        .padding()
        .background(cardBackground)
        .clipShape(RoundedRectangle(cornerRadius: 14))
        .shadow(color: .black.opacity(0.04), radius: 4, y: 2)
    }
    
    @ViewBuilder
    private var statusBadge: some View {
        switch style {
        case .overdue:
            Text("OVERDUE")
                .font(.caption.bold())
                .foregroundStyle(.white)
                .padding(.horizontal, 8)
                .padding(.vertical, 4)
                .background(.red, in: Capsule())
        case .upcoming:
            Circle()
                .fill(item.daysUntilDue <= 7 ? Color.orange : Color.blue)
                .frame(width: 12, height: 12)
        case .completed:
            Image(systemName: "checkmark.circle.fill")
                .foregroundStyle(.green)
        }
    }
    
    private var categoryColor: Color {
        switch item.category {
        case .tax: return .blue
        case .license: return .purple
        case .report: return .teal
        case .insurance: return .green
        case .employee: return .orange
        case .other: return .gray
        }
    }
    
    private var cardBackground: Color {
        switch style {
        case .overdue: return Color.red.opacity(0.06)
        case .upcoming: return Color(.systemBackground)
        case .completed: return Color.green.opacity(0.06)
        }
    }
}
```

### 3.6 Coding Rules & Conventions

| Rule | Standard |
|------|----------|
| Architecture | MVVM + Repository Pattern |
| Naming | Swift API Design Guidelines; camelCase for variables, PascalCase for types |
| Comments | `// MARK: -` for section grouping; doc comments (`///`) for public APIs |
| Error Handling | `Result<Success, Error>` for async operations; `try?` only for non-critical |
| Concurrency | `async/await` + `@MainActor` for UI updates; `Task` for fire-and-forget |
| Data Flow | SwiftData `@Query` for reads; `modelContext.insert/save` for writes |
| Notifications | Always request permission on first launch; schedule on background thread |
| Testing | XCTest for unit tests; XCUITest for critical flows; minimum 80% coverage |
| Accessibility | All interactive elements have `.accessibilityLabel()` and `.accessibilityHint()` |
| Localization | `String(localized:)` for all user-facing strings; base language: English |

### 7.6 Subscription Implementation (StoreKit 2)

```swift
import StoreKit

// MARK: - Subscription Manager
@MainActor
final class SubscriptionManager: ObservableObject {
    @Published var entitlement: Entitlement = .free
    @Published var products: [Product] = []
    
    enum Entitlement: String {
        case free = "free"
        case pro = "pro"
    }
    
    // Product IDs
    static let monthlyID = "com.deadlineguard.pro.monthly"
    static let annualID = "com.deadlineguard.pro.annual"
    static let lifetimeID = "com.deadlineguard.pro.lifetime"
    
    // Feature gates
    var maxFreeItems: Int { 3 }
    var canAddMoreItems: Bool {
        entitlement == .pro // No limit for pro
    }
    var hasMultiStageReminders: Bool {
        entitlement == .pro
    }
    var hasDocumentAttachment: Bool {
        entitlement == .pro
    }
    var hasAuditLog: Bool {
        entitlement == .pro
    }
    
    func loadProducts() async {
        do {
            let storeProducts = try await Product.products(for: [
                Self.monthlyID, Self.annualID, Self.lifetimeID
            ])
            products = storeProducts.sorted { $0.price < $1.price }
        } catch {
            print("Failed to load products: \(error)")
        }
    }
    
    func purchase(_ product: Product) async throws -> Bool {
        let result = try await product.purchase()
        switch result {
        case .success(let verification):
            if case .verified(let transaction) = verification {
                entitlement = .pro
                await transaction.finish()
                return true
            }
        case .pending, .userCancelled:
            break
        @unknown default:
            break
        }
        return false
    }
    
    func checkEntitlement() async {
        for await result in Transaction.currentEntitlements {
            if case .verified(let transaction) = result {
                entitlement = .pro
                return
            }
        }
        entitlement = .free
    }
}
```
